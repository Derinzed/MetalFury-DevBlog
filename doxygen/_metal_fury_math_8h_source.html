29
30 //convert heading to a forward vector using derived matrix
31 float x = 2 * (heading.x * heading.z + heading.w * heading.y);
32 float y = 2 * (heading.y * heading.z - heading.w * heading.x);
33 float z = 1 - 2 * (heading.x * heading.x + heading.y * heading.y);
34 MFVector3<> entityHeading(x, y, z);

78 //Solve a quadratic function
79 //Param[in] double c0: first constant
80 //Param[in] double c1: second constant
81 //Param[in] double c2: third constant
82 //Param[in] double s0: first solution
83 //Param[in] double s1: second solution
84 //Return[out] bool: true if there is a solution.
85 bool solveQuadratic(double c0, double c1, double c2, double& s0, double& s1) {
86
87 double p, q, D;
88
89 /* normal form: x^2 + px + q = 0 */
90 p = c1 / (2 * c0);
91 q = c2 / c0;
92
93 D = p * p - q;
94
95 if (D == 0.0) {
96 s0 = -p;
97 return 1;
98 }
99 else if (D < 0) {
100 return 0;
101 }
102 else /* if (D > 0) */ {
103 double sqrt_D = sqrt(D);
104
105 s0 = sqrt_D - p;
106 s1 = -sqrt_D - p;
107 return 2;
108 }
109 }

125 // Solve cubic equation: c0*x^3 + c1*x^2 + c2*x + c3.
126 // Returns number of solutions.
127 int SolveCubic(double c0, double c1, double c2, double c3, double& s0, double& s1, double& s2)
128 {
129
130 int num;
131 double sub;
132 double A, B, C;
133 double sq_A, p, q;
134 double cb_p, D;
135
136 /* normal form: x^3 + Ax^2 + Bx + C = 0 */
137 A = c1 / c0;
138 B = c2 / c0;
139 C = c3 / c0;
140
141 /* substitute x = y - A/3 to eliminate quadric term: x^3 +px + q = 0 */
142 sq_A = A * A;
143 p = 1.0 / 3 * (-1.0 / 3 * sq_A + B);
144 q = 1.0 / 2 * (2.0 / 27 * A * sq_A - 1.0 / 3 * A * B + C);
145
146 /* use Cardano's formula */
147 cb_p = p * p * p;
148 D = q * q + cb_p;
149
150 if (D == 0.0) {
151 if (q == 0.0) /* one triple solution */ {
152 s0 = 0;
153 num = 1;
154 }
155 else /* one single and one double solution */ {
156 double u = GetCubicRoot(-q);
157 s0 = 2 * u;
158 s1 = -u;
159 num = 2;
160 }
161 }
162 else if (D < 0) /* Casus irreducibilis: three real solutions */ {
163 double phi = 1.0 / 3 * acos(-q / sqrt(-cb_p));
164 double t = 2 * sqrt(-p);
165
166 s0 = t * cos(phi);
167 s1 = -t * cos(phi + 3.141592653589 / 3);
168 s2 = -t * cos(phi - 3.141592653589 / 3);
169 num = 3;
170 }
171 else /* one real solution */ {
172 double sqrt_D = sqrt(D);
173 double u = GetCubicRoot(sqrt_D - q);
174 double v = -GetCubicRoot(sqrt_D + q);
175
176 s0 = u + v;
177 num = 1;
178 }
179
180 /* resubstitute */
181 sub = 1.0 / 3 * A;
182
183 if (num > 0) s0 -= sub;
184 if (num > 1) s1 -= sub;
185 if (num > 2) s2 -= sub;
186
187 return num;
188 }

190 // Solve quartic function: c0*x^4 + c1*x^3 + c2*x^2 + c3*x + c4.
191 // Returns number of solutions.
192 int SolveQuartic(double c0, double c1, double c2, double c3, double c4, double& s0, double& s1, double& s2, double& s3) {
193
194
195 double coeffs[4] = { 0,0,0,0 };
196 double z, u, v, sub;
197 double A, B, C, D;
198 double sq_A, p, q, r;
199 int num;
200
201 /* normal form: x^4 + Ax^3 + Bx^2 + Cx + D = 0 */
202 A = c1 / c0;
203 B = c2 / c0;
204 C = c3 / c0;
205 D = c4 / c0;
206
207 /* substitute x = y - A/4 to eliminate cubic term: x^4 + px^2 + qx + r = 0 */
208 sq_A = A * A;
209 p = -3.0 / 8 * sq_A + B;
210 q = 1.0 / 8 * sq_A * A - 1.0 / 2 * A * B + C;
211 r = -3.0 / 256 * sq_A * sq_A + 1.0 / 16 * sq_A * B - 1.0 / 4 * A * C + D;
212
213 if (r == 0.0) {
214 /* no absolute term: y(y^3 + py + q) = 0 */
215
216 coeffs[3] = q;
217 coeffs[2] = p;
218 coeffs[1] = 0;
219 coeffs[0] = 1;
220
221 num = SolveCubic(coeffs[0], coeffs[1], coeffs[2], coeffs[3], s0, s1, s2);
222 }
223 else {
224 /* solve the resolvent cubic ... */
225 coeffs[3] = 1.0 / 2 * r * p - 1.0 / 8 * q * q;
226 coeffs[2] = -r;
227 coeffs[1] = -1.0 / 2 * p;
228 coeffs[0] = 1;
229
230 SolveCubic(coeffs[0], coeffs[1], coeffs[2], coeffs[3], s0, s1, s2);
231
232 /* ... and take the one real solution ... */
233 z = s0;
234
235 /* ... to build two quadric equations */
236 u = z * z - r;
237 v = 2 * z - p;
238
239 if (u == 0.0) {
240 u = 0;
241 }
242 else if (u > 0) {
243 u = sqrt(u);
244 }
245 else {
246 return 0;
247 }
248
249 if (v == 0.0) {
250 v = 0;
251 }
252 else if (v > 0) {
253 v = sqrt(v);
254 }
255 else {
256 return 0;
257 }
258
259 coeffs[2] = z - u;
260 coeffs[1] = q < 0 ? -v : v;
261 coeffs[0] = 1;
262
263 num = solveQuadratic(coeffs[0], coeffs[1], coeffs[2], s0, s1);
264
265 coeffs[2] = z + u;
266 coeffs[1] = q < 0 ? v : -v;
267 coeffs[0] = 1;
268
269 if (num == 0) num += solveQuadratic(coeffs[0], coeffs[1], coeffs[2], s0, s1);
270 else if (num == 1) num += solveQuadratic(coeffs[0], coeffs[1], coeffs[2], s1, s2);
271 else if (num == 2) num += solveQuadratic(coeffs[0], coeffs[1], coeffs[2], s2, s3);
272 }
273
274 /* resubstitute */
275 sub = 1.0 / 4 * A;
276
277 if (num > 0) s0 -= sub;
278 if (num > 1) s1 -= sub;
279 if (num > 2) s2 -= sub;
280 if (num > 3) s3 -= sub;
281
282 return num;
283 }
