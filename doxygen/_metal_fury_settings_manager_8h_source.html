20 bool loadFile(std::string filename, std::shared_ptr<Globals::ConfigurationManager::ConfigContainer> container) override {
21 //MFSettings file is a simple key-value pair file, where each line is in the format "key=value"
22 std::ifstream file(filename);
23
24 if(!file.is_open()) {
25 throw MetalFuryException("MFSettingsLoader", "Could not open file: " + filename);
26 return false;
27 }
28
29 std::string line;
30 while(std::getline(file, line)) {
31 //skip empty lines and comments
32 if(line.empty() || line[0] == '#') continue;
33 size_t delimiterPos = line.find('=');
34 if(delimiterPos == std::string::npos) continue; //invalid line format
35 std::string key = line.substr(0, delimiterPos);
36 std::string value = line.substr(delimiterPos + 1);
37 Globals::ConfigurationManager::ConfigElement element(key, value, {}, nullptr);
38 container->addElement(element);
39 }
40
41 file.close();
42 return true;
43 }
44 };
