1/* Sole is a lightweight C++11 library to generate universally unique identificators.
2 * Sole provides interface for UUID versions 0, 1 and 4.
3
4 * https://github.com/r-lyeh/sole
5 * Copyright (c) 2013,2014,2015 r-lyeh. zlib/libpng licensed.
6
7 * Based on code by Dmitri Bouianov, Philip O'Toole, Poco C++ libraries and anonymous
8 * code found on the net. Thanks guys!
9
10 * Theory: (see Hoylen's answer at [1])
11 * - UUID version 1 (48-bit MAC address + 60-bit clock with a resolution of 100ns)
12 * Clock wraps in 3603 A.D.
13 * Up to 10000000 UUIDs per second.
14 * MAC address revealed.
15 *
16 * - UUID Version 4 (122-bits of randomness)
17 * See [2] or other analysis that describe how very unlikely a duplicate is.
18 *
19 * - Use v1 if you need to sort or classify UUIDs per machine.
20 * Use v1 if you are worried about leaving it up to probabilities (e.g. your are the
21 * type of person worried about the earth getting destroyed by a large asteroid in your
22 * lifetime). Just use a v1 and it is guaranteed to be unique till 3603 AD.
23 *
24 * - Use v4 if you are worried about security issues and determinism. That is because
25 * v1 UUIDs reveal the MAC address of the machine it was generated on and they can be
26 * predictable. Use v4 if you need more than 10 million uuids per second, or if your
27 * application wants to live past 3603 A.D.
28
29 * Additionally a custom UUID v0 is provided:
30 * - 16-bit PID + 48-bit MAC address + 60-bit clock with a resolution of 100ns since Unix epoch
31 * - Format is EPOCH_LOW-EPOCH_MID-VERSION(0)|EPOCH_HI-PID-MAC
32 * - Clock wraps in 3991 A.D.
33 * - Up to 10000000 UUIDs per second.
34 * - MAC address and PID revealed.
35
36 * References:
37 * - [1] http://stackoverflow.com/questions/1155008/how-unique-is-uuid
38 * - [2] http://en.wikipedia.org/wiki/UUID#Random%5FUUID%5Fprobability%5Fof%5Fduplicates
39 * - http://en.wikipedia.org/wiki/Universally_unique_identifier
40 * - http://en.cppreference.com/w/cpp/numeric/random/random_device
41 * - http://www.itu.int/ITU-T/asn1/uuid.html f81d4fae-7dec-11d0-a765-00a0c91e6bf6
42
43 * - rlyeh ~~ listening to Hedon Cries / Until The Sun Goes up
44 */

302 inline std::string uuid::pretty() const {
303 std::stringstream ss;
304
305 uint64_t a = (ab >> 32);
306 uint64_t b = (ab & 0xFFFFFFFF);
307 uint64_t c = (cd >> 32);
308 uint64_t d = (cd & 0xFFFFFFFF);
309
310 int version = (b & 0xF000) >> 12;
311 uint64_t timestamp = ((b & 0x0FFF) << 48 ) | (( b >> 16 ) << 32) | a; // in 100ns units
312
313 ss << "version=" << (version) << ',';
314
315 if( version == 1 )
316 timestamp = timestamp - 0x01b21dd213814000ULL; // decrement Gregorian calendar
317
318 ss << std::hex << std::nouppercase << std::setfill('0');
319 version <= 1 && ss << "timestamp=" << printftime(timestamp/10000000) << ',';
320 version <= 1 && ss << "mac=" << std::setw(4) << (c & 0xFFFF) << std::setw(8) << d << ',';
321 version == 4 && ss << "randbits=" << std::setw(8) << (ab & 0xFFFFFFFFFFFF0FFFULL) << std::setw(8) << (cd & 0x3FFFFFFFFFFFFFFFULL) << ',';
322
323 ss << std::dec;
324 version == 0 && ss << "pid=" << std::setw(4) << (c >> 16 ) << ',';
325 version == 1 && ss << "clock_seq=" << std::setw(4) << ((c >> 16) & 0x3FFF) << ',';
326
327 return ss.str();
328 }

330 inline std::string uuid::str() const {
331 char uustr[] = "00000000-0000-0000-0000-000000000000";
332 constexpr char encode[] = "0123456789abcdef";
333
334 size_t bit = 15;
335 for( size_t i = 0; i < 18; i++ ) {
336 if( i == 8 || i == 13 ) {
337 continue;
338 }
339 uustr[i] = encode[ab>>4*bit&0x0f];
340 bit--;
341 }
342
343 bit = 15;
344 for( size_t i = 18; i < 36; i++ ) {
345 if( i == 18 || i == 23 ) {
346 continue;
347 }
348 uustr[i] = encode[cd>>4*bit&0x0f];
349 bit--;
350 }
351
352 return std::string(uustr);
353 }

356 int base62len = 10 + 26 + 26;
357 const char base62[] =
358 "0123456789"
359 "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
360 "abcdefghijklmnopqrstuvwxyz";
361 char res[24], *end = &res[24]; *(--end) = '\0';
362 uint64_t rem, AB = ab, CD = cd;
363 do {
364 rem = CD % base62len;
365 *--end = base62[int(rem)];
366 CD /= base62len;

368 *--end = '-';

373 } while (AB > 0);

657 inline uuid uuid1() {
658 // Number of 100-ns intervals since 00:00:00.00 15 October 1582; [ref] uuid.py
659 uint64_t ns100_intervals = get_time( 0x01b21dd213814000ULL );
660 uint16_t clock_seq = (uint16_t)( ns100_intervals & 0x3fff ); // 14-bits max
661 uint64_t mac = get_any_mac48(); // 48-bits max
662
663 uint32_t time_low = ns100_intervals & 0xffffffff;
664 uint16_t time_mid = (ns100_intervals >> 32) & 0xffff;
665 uint16_t time_hi_version = (ns100_intervals >> 48) & 0xfff;
666 uint8_t clock_seq_low = clock_seq & 0xff;
667 uint8_t clock_seq_hi_variant = (clock_seq >> 8) & 0x3f;
668
669 uuid u;
670 uint64_t &upper_ = u.ab;
671 uint64_t &lower_ = u.cd;
672
673 // Build the high 32 bytes
674 upper_ = (uint64_t) time_low << 32;
675 upper_ |= (uint64_t) time_mid << 16;
676 upper_ |= (uint64_t) time_hi_version;
677
678 // Build the low 32 bytes, using the clock sequence number
679 lower_ = (uint64_t) ((clock_seq_hi_variant << 8) | clock_seq_low) << 48;
680 lower_ |= mac;
681
682 // Set the variant to RFC 4122.
683 lower_ &= ~((uint64_t)0xc000 << 48);
684 lower_ |= (uint64_t)0x8000 << 48;
685
686 // Set the version number.
687 enum { version = 1 };
688 upper_ &= ~0xf000;
689 upper_ |= version << 12;
690
691 return u;
692 }

694 inline uuid uuid0() {
695 // Number of 100-ns intervals since Unix epoch time
696 uint64_t ns100_intervals = get_time( 0 );
697 uint64_t pid = $windows( _getpid() ) $welse( getpid() );
698 uint16_t pid16 = (uint16_t)( pid & 0xffff ); // 16-bits max
699 uint64_t mac = get_any_mac48(); // 48-bits max
700
701 uint32_t time_low = ns100_intervals & 0xffffffff;
702 uint16_t time_mid = (ns100_intervals >> 32) & 0xffff;
703 uint16_t time_hi_version = (ns100_intervals >> 48) & 0xfff;
704 uint8_t pid_low = pid16 & 0xff;
705 uint8_t pid_hi = (pid16 >> 8) & 0xff;
706
707 uuid u;
708 uint64_t &upper_ = u.ab;
709 uint64_t &lower_ = u.cd;
710
711 // Build the high 32 bytes.
712 upper_ = (uint64_t) time_low << 32;
713 upper_ |= (uint64_t) time_mid << 16;
714 upper_ |= (uint64_t) time_hi_version;
715
716 // Build the low 32 bytes, using the mac and pid number.
717 lower_ = (uint64_t) ((pid_hi << 8) | pid_low) << 48;
718 lower_ |= mac;
719
720 // Set the version number.
721 enum { version = 0 };
722 upper_ &= ~0xf000;
723 upper_ |= version << 12;
724
725 return u;
726 }

729 uuid u;
730 u.ab = ab; u.cd = cd;
731 return u;
732 }

734 inline uuid rebuild( const std::string &uustr ) {
735 uuid u = { 0, 0 };
736 auto idx = uustr.find_first_of("-");
737 if( idx != std::string::npos ) {
738 // single separator, base62 notation
739 if( uustr.find_first_of("-",idx+1) == std::string::npos ) {
740 auto rebase62 = [&]( const char *input, size_t limit ) -> uint64_t {
741 int base62len = 10 + 26 + 26;
742 auto strpos = []( char ch ) -> size_t {
743 if( ch >= 'a' ) return ch - 'a' + 10 + 26;
744 if( ch >= 'A' ) return ch - 'A' + 10;
745 return ch - '0';
746 };
747 uint64_t res = strpos( input[0] );
748 for( size_t i = 1; i < limit; ++i )
749 res = base62len * res + strpos( input[i] );
750 return res;
751 };
752 u.ab = rebase62( &uustr[0], idx );
753 u.cd = rebase62( &uustr[idx+1], uustr.size() - (idx+1) );
754 }
755 // else classic hex notation
756 else {
757 if (uustr[8] == '-' || uustr[13] == '-' || uustr[18] == '-' || uustr[23] == '-') {
758 auto decode = []( char ch ) -> size_t {
759 if( 'f' >= ch && ch >= 'a' ) return ch - 'a' + 10;
760 if( 'F' >= ch && ch >= 'A' ) return ch - 'A' + 10;
761 if( '9' >= ch && ch >= '0' ) return ch - '0';
762 return 0;
763 };
764 for( size_t i = 0; i < 18; i++ ) {
765 if( i == 8 || i == 13 ) {
766 continue;
767 }
768 u.ab = u.ab<<4 | decode(uustr[i]);
769 }
770 for( size_t i = 19; i < 36; i++ ) {
771 if( i == 23 ) {
772 continue;
773 }
774 u.cd = u.cd<<4 | decode(uustr[i]);
775 }
776 }
777 }
778 }
779 return u;
780 }
