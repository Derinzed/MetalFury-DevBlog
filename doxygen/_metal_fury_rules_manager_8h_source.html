36 bool evaluate() const {
37 switch (op) {
38 case Op::Leaf:
39 return evalLeaf();
40
41 case Op::And:
42 return left->evaluate() && right->evaluate();
43
44 case Op::Or:
45 return left->evaluate() || right->evaluate();
46
47 case Op::Not:
48 return !left->evaluate();
49 }
50 return false;
51 }
