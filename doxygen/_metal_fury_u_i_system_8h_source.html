293 // Find an element by id within the provided graph. Performs a depth-first search.
294 std::shared_ptr<UIElement> findElementById(const UIGraph& g, const std::string& id) const {
295 std::function<std::shared_ptr<UIElement>(const std::shared_ptr<UIElement>&)> recurse;
296 recurse = [&](const std::shared_ptr<UIElement>& node) -> std::shared_ptr<UIElement> {
297 if (node->properties.find("id") != node->properties.end()) {
298 if (std::holds_alternative<std::string>(node->properties.at("id").value)) {
299 if (std::get<std::string>(node->properties.at("id").value) == id) return node;
300 }
301 }
302 for (auto &c : node->children) {
303 auto found = recurse(c);
304 if (found) return found;
305 }
306 return {};
307 };
308 for (const auto &r : g.UIRoots) {
309 auto res = recurse(r);
310 if (res) return res;
311 }
312 return {};
313 }

315 // Collect all element ids present in the given graph
316 std::unordered_set<std::string> collectIds(const UIGraph& g) const {
317 std::unordered_set<std::string> ids;
318 std::function<void(const std::shared_ptr<UIElement>&)> recurse;
319 recurse = [&](const std::shared_ptr<UIElement>& node) {
320 if (node->properties.find("id") != node->properties.end()) {
321 if (std::holds_alternative<std::string>(node->properties.at("id").value)) {
322 ids.insert(std::get<std::string>(node->properties.at("id").value));
323 }
324 }
325 for (auto &c : node->children) recurse(c);
326 };
327 for (const auto &r : g.UIRoots) recurse(r);
328 return ids;
329 }
