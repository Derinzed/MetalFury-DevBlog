87 void processQueueAtLevel(int level = 1) {
88 // Process each action once, pop if not persistent, otherwise leave in queue
89 if (actions.find(level) == actions.end())
90 return;
91 auto& queue = actions[level];
92 size_t queueSize = queue.size();
93 for (size_t i = 0; i < queueSize; ++i) {
94 auto action = queue.front().get();
95 action->execute();
96 if (!action->isPersistent()) {
97 queue.pop();
98 }
99 else {
100 // Move persistent action to the back of the queue
101 queue.push(queue.front());
102 queue.pop();
103 }
104 }
