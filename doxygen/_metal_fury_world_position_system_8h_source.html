75 4. Compute zoneDelta = floor(newLocal / zoneMaxSize)
76 - floor handles negative movement so that localWithin = newLocal - zoneDelta*zoneMax is in [0, zoneMax)

79 7. If clamped:

97 // Guard against invalid zone size
98 if (zoneMaxSize <= 0.0f || zoneCount == 0u) {
99 outZone = startZone;
100 outLocal = startLocal;
101 return;
102 }
103
104 // Tentative new local position
105 float newLocal = startLocal + movement;
106
107 // Compute how many whole zones we cross (floor works for negative values)
108 float div = newLocal / zoneMaxSize;
109 long long zoneDelta = static_cast<long long>(std::floor(div));
110
111 // Local position within [0, zoneMaxSize)
112 float localWithin = newLocal - static_cast<float>(zoneDelta) * zoneMaxSize;
113
114 // Signed new zone index before clamping
115 long long newZoneSigned = static_cast<long long>(startZone) + zoneDelta;
116
117 // Clamp zone to valid range
118 long long minZone = 1;
119 long long maxZone = static_cast<long long>(zoneCount);
120
121 long long clampedZone = std::clamp(newZoneSigned, minZone, maxZone);
122
123 // If we had to clamp because we moved past the world boundary, set localWithin to an edge value
124 if (clampedZone != newZoneSigned) {
125 if (clampedZone == minZone) {
126 // At minimum world boundary -> place at local 0.0f
127 localWithin = 0.0f;
128 }
129 else {
130 // At maximum world boundary -> place just inside the max
131 localWithin = std::max(0.0f, zoneMaxSize - eps);
132 }
133 }
134 else {
135 // Ensure numerical stability: keep localWithin in [0, zoneMaxSize)
136 if (localWithin < 0.0f) localWithin = 0.0f;
137 if (localWithin >= zoneMaxSize) localWithin = std::max(0.0f, zoneMaxSize - eps);
138 }
139
140 outZone = static_cast<unsigned int>(clampedZone);
141 outLocal = localWithin;
142 };
