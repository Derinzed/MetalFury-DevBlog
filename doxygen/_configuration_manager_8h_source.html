29 void traverseXML(TiXmlElement* element, std::shared_ptr<ConfigElement> parent, std::shared_ptr<ConfigContainer> container, int depth = 0) {
30
31 if (!element) return;
32
33 //collect the element data
34 std::string innerText = element->GetText() != 0 ? element->GetText() : "";
35 std::string name = element->ValueStr();
36 std::vector<ConfigAttribute> attributes;
37 for (TiXmlAttribute* attr = element->FirstAttribute(); attr; attr = attr->Next()) {
38 attributes.push_back(ConfigAttribute(attr->Name(), attr->Value()));
39 }
40
41 ConfigElement newElement(name, innerText, attributes, parent);
42 container->addElement(newElement);
43
44 // Recursively iterate through child elements
45 auto child = element->FirstChildElement();
46 if (child) {

48 }

52 if (sibling){
